Create `app2/src/app/core/services/sqlite.service.ts`:
```typescript
import { Injectable, signal } from '@angular/core';
import { Capacitor } from '@capacitor/core';
import { CapacitorSQLite, SQLiteConnection, SQLiteDBConnection } from '@capacitor-community/sqlite';

const DB_NAME = 'moneymanager';

@Injectable({ providedIn: 'root' })
export class SqliteService {
  private sqlite = new SQLiteConnection(CapacitorSQLite);
  private db!: SQLiteDBConnection;
  readonly ready = signal(false);

  async init(): Promise<void> {
    if (Capacitor.getPlatform() === 'web') {
      await customElements.whenDefined('jeep-sqlite');
      await this.sqlite.initWebStore();
    }
    this.db = await this.sqlite.createConnection(DB_NAME, false, 'no-encryption', 1, false);
    await this.db.open();
    await this.createTables();
    this.ready.set(true);
  }

  private async createTables(): Promise<void> {
    await this.db.execute(`
      CREATE TABLE IF NOT EXISTS users (
        id TEXT PRIMARY KEY,
        username TEXT NOT NULL,
        balance REAL DEFAULT 0,
        settings TEXT DEFAULT '{}',
        language TEXT DEFAULT 'de',
        created TEXT,
        updated TEXT
      );
      CREATE TABLE IF NOT EXISTS contacts (
        id TEXT PRIMARY KEY,
        name TEXT NOT NULL,
        linkedName TEXT DEFAULT '',
        balance REAL DEFAULT 0,
        owner TEXT NOT NULL,
        user TEXT DEFAULT '',
        statistics TEXT DEFAULT '',
        score REAL DEFAULT 0,
        created TEXT,
        updated TEXT,
        synced INTEGER DEFAULT 0
      );
      CREATE TABLE IF NOT EXISTS transactions (
        id TEXT PRIMARY KEY,
        amount REAL NOT NULL,
        info TEXT DEFAULT '',
        date TEXT NOT NULL,
        type TEXT NOT NULL,
        contact TEXT NOT NULL,
        owner TEXT NOT NULL,
        courierLink TEXT DEFAULT '',
        planned INTEGER DEFAULT 0,
        created TEXT,
        updated TEXT,
        synced INTEGER DEFAULT 0,
        FOREIGN KEY (contact) REFERENCES contacts(id)
      );
      CREATE TABLE IF NOT EXISTS statistics (
        id TEXT PRIMARY KEY,
        balanceHistory TEXT DEFAULT '[]',
        created TEXT,
        updated TEXT
      );
    `);
  }

  async query<T>(sql: string, params: any[] = []): Promise<T[]> {
    const result = await this.db.query(sql, params);
    return (result.values ?? []) as T[];
  }

  async run(sql: string, params: any[] = []): Promise<void> {
    await this.db.run(sql, params);
  }

  async getById<T>(table: string, id: string): Promise<T | null> {
    const rows = await this.query<T>(`SELECT * FROM ${table} WHERE id = ?`, [id]);
    return rows[0] ?? null;
  }

  async getAll<T>(table: string, orderBy = 'created DESC'): Promise<T[]> {
    return this.query<T>(`SELECT * FROM ${table} ORDER BY ${orderBy}`);
  }

  async upsert(table: string, data: Record<string, any>): Promise<void> {
    const keys = Object.keys(data);
    const placeholders = keys.map(() => '?').join(',');
    const updates = keys.map(k => `${k}=excluded.${k}`).join(',');
    const values = Object.values(data);
    await this.run(
      `INSERT INTO ${table} (${keys.join(',')}) VALUES (${placeholders}) ON CONFLICT(id) DO UPDATE SET ${updates}`,
      values
    );
  }

  async delete(table: string, id: string): Promise<void> {
    await this.run(`DELETE FROM ${table} WHERE id = ?`, [id]);
  }
}

import { Injectable, signal } from '@angular/core';
import PocketBase from 'pocketbase';

@Injectable({ providedIn: 'root' })
export class PocketbaseService {
  private pb = new PocketBase('http://localhost:8090');
  readonly online = signal(false);

  constructor() {
    this.pb.autoCancellation(false);
    this.checkConnection();
  }

  get client(): PocketBase {
    return this.pb;
  }

  get userId(): string | undefined {
    return this.pb.authStore.record?.id;
  }

  get isAuthenticated(): boolean {
    return this.pb.authStore.isValid;
  }

  async checkConnection(): Promise<void> {
    try {
      await this.pb.health.check();
      this.online.set(true);
    } catch {
      this.online.set(false);
    }
  }

  async login(username: string, password: string) {
    const result = await this.pb.collection('users').authWithPassword(username, password);
    this.online.set(true);
    return result;
  }

  async register(username: string, password: string) {
    const user = await this.pb.collection('users').create({
      username,
      password,
      passwordConfirm: password,
      balance: 0,
    });
    return this.login(username, password);
  }

  logout(): void {
    this.pb.authStore.clear();
  }
}
```

- [ ] **Step 2: Create Auth service**

Create `app2/src/app/core/services/auth.service.ts`:
```typescript
import { Injectable, signal, computed } from '@angular/core';
import { Router } from '@angular/router';
import { PocketbaseService } from './pocketbase.service';
import { SqliteService } from './sqlite.service';
import type { User } from '../models/user.model';

@Injectable({ providedIn: 'root' })
export class AuthService {
  readonly user = signal<User | null>(null);
  readonly isAuthenticated = computed(() => !!this.user());

  constructor(
    private pb: PocketbaseService,
    private sqlite: SqliteService,
    private router: Router,
  ) {}

  async init(): Promise<void> {
    if (this.pb.isAuthenticated) {
      const record = this.pb.client.authStore.record;
      if (record) {
        const user: User = {
          id: record['id'],
          username: record['username'],
          balance: record['balance'] ?? 0,
          settings: record['settings'] ?? {},
          language: record['language'] ?? 'de',
          created: record['created'],
          updated: record['updated'],
        };
        await this.sqlite.upsert('users', user);
        this.user.set(user);
      }
    }
  }

  async login(username: string, password: string): Promise<void> {
    const { record } = await this.pb.login(username, password);
    const user: User = {
      id: record['id'],
      username: record['username'],
      balance: record['balance'] ?? 0,
      settings: record['settings'] ?? {},
      language: record['language'] ?? 'de',
      created: record['created'],
      updated: record['updated'],
    };
    await this.sqlite.upsert('users', user);
    this.user.set(user);
  }

  async register(username: string, password: string): Promise<void> {
    await this.pb.register(username, password);
    await this.login(username, password);
  }

  logout(): void {
    this.pb.logout();
    this.user.set(null);
    this.router.navigate(['/auth/login']);
  }
}
```

import { SqliteService } from '../../../core/services/sqlite.service';
import { AuthService } from '../../../core/services/auth.service';
import type { Contact } from '../../../core/models/contact.model';

@Injectable({ providedIn: 'root' })
export class ContactService {
  readonly contacts = signal<Contact[]>([]);

  readonly owned = computed(() =>
    this.contacts().filter(c => c.owner === this.auth.user()?.id && !c.user)
  );
  readonly linked = computed(() =>
    this.contacts().filter(c => c.user === this.auth.user()?.id && c.owner !== this.auth.user()?.id)
  );

  constructor(private sqlite: SqliteService, private auth: AuthService) {}

  async loadAll(): Promise<void> {
    const rows = await this.sqlite.getAll<Contact>('contacts', 'name ASC');
    this.contacts.set(rows);
  }

  async getById(id: string): Promise<Contact | null> {
    return this.sqlite.getById<Contact>('contacts', id);
  }

  async create(name: string, user?: string): Promise<void> {
    const id = crypto.randomUUID().replace(/-/g, '').slice(0, 15);
    const now = new Date().toISOString();
    const contact: Contact = {
      id,
      name,
      linkedName: '',
      balance: 0,
      owner: this.auth.user()!.id,
      user: user ?? '',
      statistics: '',
      score: 0,
      created: now,
      updated: now,
      synced: false,
    };
    await this.sqlite.upsert('contacts', { ...contact, synced: 0 });
    this.contacts.update(list => [...list, contact]);
  }

  async update(id: string, data: Partial<Contact>): Promise<void> {
    const updated = { ...data, id, updated: new Date().toISOString(), synced: 0 };
    await this.sqlite.upsert('contacts', updated);
    this.contacts.update(list => list.map(c => c.id === id ? { ...c, ...data } : c));
  }

  async remove(id: string): Promise<void> {
    await this.sqlite.delete('contacts', id);
    this.contacts.update(list => list.filter(c => c.id !== id));
  }
}

import { Injectable, signal } from '@angular/core';
import { SqliteService } from '../../../core/services/sqlite.service';
import { AuthService } from '../../../core/services/auth.service';
import { Transaction, TransactionType } from '../../../core/models/transaction.model';

@Injectable({ providedIn: 'root' })
export class TransactionService {
  readonly transactions = signal<Transaction[]>([]);

  constructor(private sqlite: SqliteService, private auth: AuthService) {}

  async loadByContact(contactId: string, limit = 50): Promise<Transaction[]> {
    return this.sqlite.query<Transaction>(
      'SELECT * FROM transactions WHERE contact = ? AND planned = 0 ORDER BY date DESC LIMIT ?',
      [contactId, limit]
    );
  }

  async loadRecent(limit = 20): Promise<Transaction[]> {
    const userId = this.auth.user()?.id;
    return this.sqlite.query<Transaction>(
      'SELECT * FROM transactions WHERE owner = ? AND planned = 0 ORDER BY date DESC LIMIT ?',
      [userId, limit]
    );
  }

  async loadPlanned(): Promise<Transaction[]> {
    const userId = this.auth.user()?.id;
    return this.sqlite.query<Transaction>(
      'SELECT * FROM transactions WHERE owner = ? AND planned = 1 ORDER BY date DESC',
      [userId]
    );
  }

  async loadByTimeframe(contactId: string, startDate: string): Promise<Transaction[]> {
    return this.sqlite.query<Transaction>(
      'SELECT * FROM transactions WHERE contact = ? AND planned = 0 AND date >= ? ORDER BY date DESC',
      [contactId, startDate]
    );
  }

  async create(data: { amount: number; type: TransactionType; contact: string; info?: string; planned?: boolean }): Promise<Transaction> {
    const id = crypto.randomUUID().replace(/-/g, '').slice(0, 15);
    const now = new Date().toISOString();
    const tx: Transaction = {
      id,
      amount: data.amount,
      info: data.info ?? '',
      date: now,
      type: data.type,
      contact: data.contact,
      owner: this.auth.user()!.id,
      courierLink: '',
      planned: data.planned ?? false,
      created: now,
      updated: now,
      synced: false,
    };
    await this.sqlite.upsert('transactions', { ...tx, planned: tx.planned ? 1 : 0, synced: 0 });

    if (!tx.planned) {
      await this.updateContactBalance(tx);
    }

    return tx;
  }

  async confirmPlanned(id: string): Promise<void> {
    const now = new Date().toISOString();
    await this.sqlite.run(
      'UPDATE transactions SET planned = 0, date = ?, updated = ?, synced = 0 WHERE id = ?',
      [now, now, id]
    );
    const tx = await this.sqlite.getById<Transaction>('transactions', id);
    if (tx) await this.updateContactBalance({ ...tx, planned: false });
  }

  async remove(id: string): Promise<void> {
    const tx = await this.sqlite.getById<Transaction>('transactions', id);
    if (tx && !tx.planned) await this.updateContactBalance(tx, true);
    await this.sqlite.delete('transactions', id);
  }

  private async updateContactBalance(tx: Transaction, reverse = false): Promise<void> {
    const mult = reverse ? -1 : 1;
    let delta = 0;
    if (tx.type === TransactionType.Invoice) delta = -tx.amount * mult;
    if (tx.type === TransactionType.Refund) delta = tx.amount * mult;

    if (delta !== 0) {
      await this.sqlite.run(
        'UPDATE contacts SET balance = balance + ?, updated = ?, synced = 0 WHERE id = ?',
        [delta, new Date().toISOString(), tx.contact]
      );
    }
  }
}

export type Timeframe = '1w' | '1m' | '3m' | '6m' | '1y' | 'max';

export function getStartDate(tf: Timeframe): Date | null {
  if (tf === 'max') return null;
  const d = new Date();
  switch (tf) {
    case '1w': d.setDate(d.getDate() - 7); break;
    case '1m': d.setMonth(d.getMonth() - 1); break;
    case '3m': d.setMonth(d.getMonth() - 3); break;
    case '6m': d.setMonth(d.getMonth() - 6); break;
    case '1y': d.setFullYear(d.getFullYear() - 1); break;
  }
  return d;
}

export class BalanceGraphComponent implements AfterViewInit {
  data = input.required<BalancePoint[]>();
  width = 320;
  height = 120;
  svg = viewChild<ElementRef<SVGElement>>('svg');

  constructor() {
    effect(() => {
      this.render(this.data());
    });
  }

  ngAfterViewInit() {
    this.render(this.data());

  private render(points: BalancePoint[]) {
    const el = this.svg()?.nativeElement;
    if (!el || points.length < 2) {
      if (el) el.innerHTML = '<text x="160" y="60" fill="#666" font-size="14" text-anchor="middle">Keine Daten</text>';
      return;

    const h = this.height - padding.top - padding.bottom;
    const balances = points.map(p => p.balance);
    const min = Math.min(...balances);
    const max = Math.max(...balances);
    const range = max - min || 1;

      x: padding.left + (i / (points.length - 1)) * w,
      y: padding.top + h - ((p.balance - min) / range) * h,
    }));

    const polyline = coords.map(c => `${c.x},${c.y}`).join(' ');
    const areaPath = `M${coords[0].x},${coords[0].y} ${coords.slice(1).map(c => `L${c.x},${c.y}`).join(' ')} L${coords[coords.length - 1].x},${padding.top + h} L${coords[0].x},${padding.top + h} Z`;
    const color = balances[balances.length - 1] >= 0 ? '#4cd964' : '#ff3b30';

```typescript
import { Injectable, signal } from '@angular/core';
import { SqliteService } from './sqlite.service';
import { PocketbaseService } from './pocketbase.service';

@Injectable({ providedIn: 'root' })
export class SyncService {
  readonly syncing = signal(false);
  readonly lastSync = signal<string | null>(null);

  constructor(private sqlite: SqliteService, private pb: PocketbaseService) {}

  async syncAll(): Promise<void> {
    if (!this.pb.online() || !this.pb.isAuthenticated || this.syncing()) return;
    this.syncing.set(true);

    try {
      await this.pushLocal('contacts');
      await this.pushLocal('transactions');
      await this.pullRemote('contacts');
      await this.pullRemote('transactions');
      this.lastSync.set(new Date().toISOString());
    } finally {
      this.syncing.set(false);
    }
  }

  private async pushLocal(collection: string): Promise<void> {
    const unsynced = await this.sqlite.query<any>(
      `SELECT * FROM ${collection} WHERE synced = 0`
    );
    for (const record of unsynced) {
      try {
        const { synced, ...data } = record;
        try {
          await this.pb.client.collection(collection).update(record.id, data);
        } catch {
          await this.pb.client.collection(collection).create({ ...data, id: record.id });
        }
        await this.sqlite.run(
          `UPDATE ${collection} SET synced = 1 WHERE id = ?`,
          [record.id]
        );
      } catch (e) {
        console.error(`Sync push failed for ${collection}/${record.id}:`, e);
      }
    }
  }

  private async pullRemote(collection: string): Promise<void> {
    try {
      const lastSync = this.lastSync();
      let filter = '';
      if (lastSync) {
        filter = `updated > "${lastSync}"`;
      }
      const records = await this.pb.client.collection(collection).getFullList({
        filter,
        sort: '-updated',
      });
      for (const record of records) {
        const data: any = { ...record, synced: 1 };
        delete data.expand;
        delete data.collectionId;
        delete data.collectionName;
        await this.sqlite.upsert(collection, data);
      }
    } catch (e) {
      console.error(`Sync pull failed for ${collection}:`, e);
    }
  }

  startPeriodicSync(intervalMs = 30000): void {
    setInterval(() => this.syncAll(), intervalMs);
  }
}
```
